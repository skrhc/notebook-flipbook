    <script>

        const totalPages = 110;

        let currentPage = 1;


        const imgLeft = document.getElementById('img-left');

        const imgRight = document.getElementById('img-right');

        const pageNum = document.getElementById('page-num');

        const prevBtn = document.getElementById('prev-btn');

        const nextBtn = document.getElementById('next-btn');

        const clickLeft = document.querySelector('.click-left');

        const clickRight = document.querySelector('.click-right');


        function padNumber(num) {

            return String(num).padStart(3, '0');

        }


        function updatePages() {

            const leftPage = currentPage;

            const rightPage = currentPage + 1;


            imgLeft.src = `book/page-${padNumber(leftPage)}.jpg`;

            

            if (rightPage <= totalPages) {

                imgRight.src = `book/page-${padNumber(rightPage)}.jpg`;

                imgRight.style.display = 'block';

            } else {

                imgRight.style.display = 'none';

            }


            if (rightPage <= totalPages) {

                pageNum.textContent = `Pages ${leftPage}-${rightPage} of ${totalPages}`;

            } else {

                pageNum.textContent = `Page ${leftPage} of ${totalPages}`;

            }


            prevBtn.disabled = currentPage <= 1;

            nextBtn.disabled = currentPage >= totalPages - 1;

        }


        function nextPage() {

            if (currentPage < totalPages - 1) {

                currentPage += 2;

                updatePages();

            }

        }


        function prevPage() {

            if (currentPage > 1) {

                currentPage -= 2;

                updatePages();

            }

        }


        // Event listeners

        nextBtn.addEventListener('click', nextPage);

        prevBtn.addEventListener('click', prevPage);

        clickRight.addEventListener('click', nextPage);

        clickLeft.addEventListener('click', prevPage);


        // Keyboard navigation

        document.addEventListener('keydown', (e) => {

            if (e.key === 'ArrowRight') nextPage();

            if (e.key === 'ArrowLeft') prevPage();

        });


        // Initialize

        updatePages();

    </script>
